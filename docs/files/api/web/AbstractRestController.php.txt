require_once ('../../Constants.php');

/**
 * Class AbstractRestController
 * @abstract
 */
class AbstractRestController
{
    /**
     * @param $content
     */
    public static function flushJSONContent($content){
        ob_start();
        header(HEADER_JSON);
        echo(json_encode($content));
        ob_end_flush();
    }

}
